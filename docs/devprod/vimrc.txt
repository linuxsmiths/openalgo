" The commands in this are executed when vim/gvim is started.
"
" To use it, copy it to
"     for Unix and OS/2:  ~/.gvimrc
"             for Amiga:  s:.gvimrc
"  for MS-DOS and Win32:  $VIM\_gvimrc

" Use Vim settings, rather then Vi settings (much better!).
" This must be first, because it changes other options as a side effect.
set nocompatible

" Make status/command line two lines high
set ch=2

" Make shift-insert work like in Xterm
map <S-Insert> <MiddleMouse>
map! <S-Insert> <MiddleMouse>

" allow backspacing over everything in insert mode
set bs=2

" always set autoindenting on
set ai
set ci

" do not keep a backup file, use versions instead
set nobackup

" read/write a .viminfo file, don't store more than 50 lines of registers
set viminfo='20,\"50
" keep 50 lines of command line history
set history=50
" show the cursor position all the time
set ruler

" Switch on syntax highlighting.
syntax on
" Switch on search pattern highlighting.
set hlsearch
" Hide the mouse pointer while typing
set mousehide

" Make p in Visual mode replace the selected text with the "" register.
vnoremap p <Esc>:let current_reg = @"<CR>gvdi<C-R>=current_reg<CR><Esc>

"auto wrap line at 100 columns
set tw=100
set colorcolumn=100

if has("gui_running")
    " set the X11 font to use
    set guifont=Monospace\ 18
    colorscheme desert
endif

set showmode
set noerrorbells

"text should wrap around the gvim window
set wrap
set autowrite

" Set tab stops
set ts=4
" Number of spaces to use for (auto)indent step
set sw=4
set tags=tags

"tab expands are not good in Makefile.
if argv(0) != "Makefile"
   set expandtab
else
   set noexpandtab
endif

set formatoptions=croql
" the following causes Ctrl-T to show the tags list in case of multiple tags,
" instead of jumping to the first one.
set cscopetag

"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"  Below this lines are personalized defines which you can change.
"
"""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""""
"gf search path
set path=.,/usr/include,/usr/src/linux/include
set tags=tags

"Using vim-plug
call plug#begin('~/.vim/plugged')
Plug 'github/copilot.vim'
call plug#end()
